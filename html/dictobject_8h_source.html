00016 /*
00017 There are three kinds of slots in the table:
00018 
00019 1. Unused.  me_key == me_value == NULL
00020    Does not hold an active (key, value) pair now and never did.  Unused can
00021    transition to Active upon key insertion.  This is the only case in which
00022    me_key is NULL, and is each slot's initial state.
00023 
00024 2. Active.  me_key != NULL and me_key != dummy and me_value != NULL
00025    Holds an active (key, value) pair.  Active can transition to Dummy upon
00026    key deletion.  This is the only case in which me_value != NULL.
00027 
00028 3. Dummy.  me_key == dummy and me_value == NULL
00029    Previously held an active (key, value) pair, but that was deleted and an
00030    active pair has not yet overwritten the slot.  Dummy can transition to
00031    Active upon key insertion.  Dummy slots cannot be made Unused again
00032    (cannot have me_key set to NULL), else the probe sequence in case of
00033    collision would have no way to know they were once active.
00034 
00035 Note: .popitem() abuses the me_hash field of an Unused or Dummy slot to
00036 hold a search finger.  The me_hash field of Unused or Dummy slots has no
00037 meaning otherwise.
00038 */

00040 /* PyDict_MINSIZE is the minimum size of a dictionary.  This many slots are
00041  * allocated directly in the dict object (in the ma_smalltable member).
00042  * It must be a power of 2, and at least 4.  8 allows dicts with no more
00043  * than 5 active entries to live in ma_smalltable (and so avoid an
00044  * additional malloc); instrumentation suggested this suffices for the
00045  * majority of dicts (consisting mostly of usually-small instance dicts and
00046  * usually-small dicts created to pass keyword arguments).
00047  */
00048 #define PyDict_MINSIZE 8

00060 /*
00061 To ensure the lookup algorithm terminates, there must be at least one Unused
00062 slot (NULL key) in the table.
00063 The value ma_fill is the number of non-NULL keys (sum of Active and Dummy);
00064 ma_used is the number of non-NULL, non-dummy keys (== the number of non-NULL
00065 values == the number of Active items).
00066 To avoid slowing down lookups on a near-full table, we resize the table when
00067 it's two-thirds full.
00068 */
00069 typedef struct _dictobject PyDictObject;
00070 struct _dictobject {
00071     PyObject_HEAD
00072     Py_ssize_t ma_fill;  /* # Active + # Dummy */
00073     Py_ssize_t ma_used;  /* # Active */
00074 
00075     /* The table contains ma_mask + 1 slots, and that's a power of 2.
00076      * We store the mask instead of the size because the mask is more
00077      * frequently needed.
00078      */
00079     Py_ssize_t ma_mask;
00080 
00081     /* ma_table points to ma_smalltable for small tables, else to
00082      * additional malloc'ed memory.  ma_table is never NULL!  This rule
00083      * saves repeated runtime null-tests in the workhorse getitem and
00084      * setitem calls.
00085      */
00086     PyDictEntry *ma_table;
00087     PyDictEntry *(*ma_lookup)(PyDictObject *mp, PyObject *key, long hash);
00088     PyDictEntry ma_smalltable[PyDict_MINSIZE];
