00012 /*
00013 Type PyStringObject represents a character string.  An extra zero byte is
00014 reserved at the end to ensure it is zero-terminated, but a size is
00015 present so strings with null bytes in them can be represented.  This
00016 is an immutable object type.
00017 
00018 There are functions to create new string objects, to test
00019 an object for string-ness, and to get the
00020 string value.  The latter function returns a null pointer
00021 if the object is not of the proper type.
00022 There is a variant that takes an explicit size as well as a
00023 variant that assumes a zero-terminated string.  Note that none of the
00024 functions should be applied to nil objects.

00027 /* Caching the hash (ob_shash) saves recalculation of a string's hash value.
00028    Interning strings (ob_sstate) tries to ensure that only one string
00029    object with a given value exists, so equality tests can be one pointer
00030    comparison.  This is generally restricted to strings that "look like"
00031    Python identifiers, although the intern() builtin can be used to force
00032    interning of any string.
00033    Together, these sped the interpreter by up to 20%. */
00034 
00035 typedef struct {
00036     PyObject_VAR_HEAD
00037     long ob_shash;
00038     int ob_sstate;
00039     char ob_sval[1];
00040 
00041     /* Invariants:
00042      *     ob_sval contains space for 'ob_size+1' elements.
00043      *     ob_sval[ob_size] == 0.
00044      *     ob_shash is the hash of the string or -1 if not computed yet.
00045      *     ob_sstate != 0 iff the string object is in stringobject.c's
00046      *       'interned' dictionary; in this case the two references
00047      *       from 'interned' to this object are *not counted* in ob_refcnt.
00048      */
00049 } PyStringObject;
